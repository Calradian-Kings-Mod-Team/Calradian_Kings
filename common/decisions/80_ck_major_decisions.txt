				}

			}
			desc = form_transtanaesic_kingdom_decision
		}
	}
	desc = {

						highest_held_title_tier = tier_empire
					}
				}
				desc = form_transtanaesic_kingdom_decision_desc.calradian
			}
			triggered_desc = {

				}

			}
			triggered_desc = {

			}
			desc = form_transtanaesic_kingdom_decision_desc
		}
	}

	confirm_text = {
		first_valid = {
			triggered_desc = {
				trigger = { 
					culture = { has_cultural_pillar = heritage_calradian }
					top_liege = {
						culture = { has_cultural_pillar = heritage_calradian }
						highest_held_title_tier = tier_empire
					}
				}
				desc = form_transtanaesic_kingdom_decision_confirm.calradian
			}
			triggered_desc = {
				trigger = { 
					culture = { has_cultural_pillar = heritage_kohinik }
				}
				desc = form_transtanaesic_kingdom_decision_confirm.kohinik
			}
			desc = form_transtanaesic_kingdom_decision_confirm
		}
	}
	
	selection_tooltip = form_transtanaesic_kingdom_decision_tooltip

	ai_check_interval = 120

	is_shown = {
		is_landed = yes
		NOT = { exists = title:k_transtanaesia.holder }
		NOT = {  #Can only do it once.
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:flag_formed_transtanaesia_kingdom
			}
		}
		#No kings or emperors.
		highest_held_title_tier <= tier_duchy
		NOT = {
			culture = { has_cultural_pillar = heritage_khuzic }
		}
		any_sub_realm_county = {
			title_province = { geographical_region = region_k_tanaesis }
		}
		NOT = {
			culture = { has_cultural_pillar = heritage_khuzic }
		}
	}

	is_valid = {
		OR = {
			is_independent_ruler = yes
			custom_description = {
				text = vassal_of_a_calradian_empire
				trigger_if = {
					limit = {
						AND = {
							top_liege = {
								AND = {
									culture = { has_cultural_pillar = heritage_calradian }
									highest_held_title_tier = tier_empire
								}
							}
						}
					}
				}
				
			}
			
		}
		
		prestige_level >= high_prestige_level
		#And the core of Tanaesis
		completely_controls_region = region_k_tanaesis
	}

	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_at_war = no
	}

	cost = {
		gold = 200
	}

	effect = {
		if = {
			limit = {
				culture = { has_cultural_pillar = heritage_calradian }
			}
			save_scope_as = calradian_founder
		}
		else_if = {
			limit = {
				culture = { has_cultural_pillar = heritage_kohinik }
			}
			save_scope_as = kohinik_founder
		}
		gain_heroic_legend_seed_tooltip_effect = yes
		# show_as_tooltip = {
		# 	form_switzerland_scripted_effect = yes
		# }
		trigger_event = ck_tanesis_events.0001

		hidden_effect = {
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:flag_formed_transtanaesia_kingdom
			}
			set_global_variable = {
				name = flag_formed_transtanaesia_kingdom
				value = root
			}
		}
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}

## Rhodok Council Decision ##
rhodok_council_decision = {
	ai_check_interval = 0
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_recruitment.dds"
	}
	decision_group_type = major
	desc = rhodok_council_decision_desc
	
	is_shown = {
		is_ai = no
		is_landed = yes
		is_vassal_or_below_of = title:k_vlandia.holder
		culture:rhodok = {
			has_variable = rhodok_council_possibility_var
		}
		culture = {
			OR = {
				this = culture:rhodok
				any_parent_culture_or_above = {
					this = culture:rhodok
				}
			}
		}
	}
	
	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_at_war = no
		is_in_an_activity = no
	}

	is_valid = {
		custom_tooltip = {
			text = rhodok_culture_tt
			culture = {
				OR = {
					this = culture:rhodok
					any_parent_culture_or_above = {
						this = culture:rhodok
					}
				}
			}
		}
		custom_tooltip = {
			text = is_vassal_of_vlandian_king_tt
			is_vassal_or_below_of = title:k_vlandia.holder
		}
		capital_province = { geographical_region = region_rhodok_culture }
		NOT = {
			has_trait = loyal
		}
		prestige_level >= 1
	}

	effect = {
		custom_tooltip = {
			text = rhodok_council_decision_tt
			culture:rhodok = {
				remove_variable = rhodok_council_possibility_var
			}
		}
	}
	
	
	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = 0
	}
}
