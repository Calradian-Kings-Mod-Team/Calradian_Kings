	}
	decision_group_type = major
	desc = rhodok_council_decision_desc
	
	is_shown = {
		is_ai = no
		is_landed = yes
		is_vassal_or_below_of = title:k_vlandia.holder
		NOR = {
			has_character_flag = rejected_rhodok_council
			has_character_flag = rejected_rhodok_war
		}
		culture:rhodok = {
			has_variable = rhodok_council_possibility_var
		}
		culture:rhodok = {
			NOT = {
				has_variable = independence_war_available_rk
			}
		}
		culture = {
			OR = {
				this = culture:rhodok
				any_parent_culture_or_above = {
					this = culture:rhodok
				}
			}
		}
	}
	
	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_at_war = no
		NOT = { exists = involved_activity }
		custom_tooltip = {
			text = rhodok_council_tt
			culture:rhodok = {
				NOT = {
					has_variable = rhodok_council_established
				}
			}
		}
		custom_tooltip = {
			text = rhodok_instigiator_died
			culture = {
				NOT = {
					has_variable = council_invitation_cooldown
				}
			}
		}
	}

	is_valid = {
		is_adult = yes
		custom_tooltip = {
			text = rhodok_culture_tt
			culture = {
				OR = {
					this = culture:rhodok
					any_parent_culture_or_above = {
						this = culture:rhodok
					}
				}
			}
		}
		custom_tooltip = {
			text = rhodok_council_not_tt
			culture:rhodok = {
				NOT = {
					has_variable = rhodok_council_established
				}
			}
		}
		custom_tooltip = {
			text = is_vassal_of_vlandian_king_tt
			is_vassal_or_below_of = title:k_vlandia.holder
		}
		capital_province = { geographical_region = region_rhodok_culture }
		NOT = {
			has_trait = loyal
		}
		prestige_level >= 1
	}

	effect = {
		save_scope_as = rhodok_instigator
		add_character_flag = rhodok_instigator_flag
		custom_tooltip = {
			text = rhodok_council_decision_tt
			remove_rhodok_council_var_effect = yes
			culture:rhodok = {
				set_variable = rhodok_council_established
			}
			trigger_event = {
				id = ck_global_culture.0018
				days = 30
			}
		}
		hidden_effect = {
			send_rhodok_council_invitation_effect = yes
			send_rhodok_council_invitation_swadian_effect = yes
		}
	}
	
	
	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = 0
	}
}
