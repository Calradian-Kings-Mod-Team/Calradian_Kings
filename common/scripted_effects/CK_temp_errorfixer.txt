fix_variable_error = {# ERROR = KEY
	if = {
		limit = {
			exists = var:$ERROR$
			var:$ERROR$ = flag:$ERROR$
		}
		set_variable = {
			name = $ERROR$
			value = flag:$ERROR$
		}
	}
}
fix_scope_error = {
	if = {
		limit = {
			exists = scope:$ERROR$
		}
		save_scope_as = $ERROR$
	}
}
