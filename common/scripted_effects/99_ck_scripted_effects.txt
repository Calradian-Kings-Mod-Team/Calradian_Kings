	if = {
		limit = {
			NOT = {
				culture = {
					has_cultural_pillar = heritage_calradian
				}
			}

				percent > 0.5
				culture = {
					has_cultural_pillar = heritage_calradian
				}
			}
		}

	if = {
		limit = {
			faith = {
				NOR = {
					has_doctrine_parameter = unlock_jizya_contract

				}
			}
			any_vassal = {
				vassal_contract_obligation_level_can_be_decreased = jizya_special_rights
			}
		}

				vassal_contract_obligation_level_can_be_decreased = jizya_special_rights
			}

	}
