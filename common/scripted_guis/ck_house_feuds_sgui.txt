ck_house_feuds_visible = {
	scope = character

	is_shown = {
		culture = {
			has_cultural_pillar = heritage_nahasi
		}
	}
}

ck_feud_window = {
	scope = character
	effect = {
        if = {
            limit = {
                has_variable = show_feud_window
            }
            remove_variable = show_feud_window
        }
        else = {
            set_variable = show_feud_window
        }
    }
}
