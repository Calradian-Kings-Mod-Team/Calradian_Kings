culture_ai_weight_skirmishers = {
	value = -10
	
	if = { # Cultures that favours skirmishers
		limit = {
			culture = {
				OR = {
					has_cultural_pillar = heritage_nahasi
					has_cultural_pillar = heritage_chertygic
					has_cultural_pillar = heritage_darshi
				}
			}
		}
		add = 20
	}
	if = { # Cultures that dislike skirmishers
		limit = {
			culture = {
				OR = {
					has_cultural_pillar = heritage_calradian
					has_cultural_pillar = heritage_balian
					has_cultural_pillar = heritage_byalic
					has_cultural_pillar = heritage_palaic
				}
			}
		}
		add = -60
	}
	
	if = { # Cultures with a special MaA of that type should avoid the base version of the unit
		limit = {
			culture = {
				culture_has_skirmisher_maa = yes
			}
		}
		add = -500
	}
}
