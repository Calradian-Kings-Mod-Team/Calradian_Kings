	scope:vassal = {
		if = {
			limit = {
				AND = {
					scope:liege ?= {
						AND = {
							highest_held_title_tier >= tier_kingdom
							culture = {
								has_cultural_pillar = heritage_sturgic
							}
							primary_title = {
								NOT = {
									has_title_law_flag = elective_succession_law
								}
							}
						}
						
					}
					scope:vassal.culture = {
						has_cultural_parameter = unruly_vaegirs
					}
				}
			}
			value = -30
		}
		else_if = {
			limit = {
				AND = {
					scope:liege ?= {
						AND = {
							highest_held_title_tier >= tier_kingdom
							culture = {
								has_cultural_pillar = heritage_sturgic
							}
							primary_title = {
								has_title_law_flag = elective_succession_law
							}
						}
						
					}
					scope:vassal.culture = {
						has_cultural_parameter = unruly_vaegirs
					}
				}
			}
			value = 15
		}
	}
}
