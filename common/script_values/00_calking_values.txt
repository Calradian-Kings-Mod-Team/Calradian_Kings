sturgian_elective_opinion_value = {
	scope:vassal = {
		if = {
			limit = {
				AND = {
					scope:liege ?= {
						highest_held_title_tier >= tier_kingdom
					}
					scope.liege.culture = {
						has_cultural_pillar = heritage_byalic
					}
					scope:liege.primary_title = {
						NOT = {
							has_title_law_flag = elective_succession_law
						}
					}
					scope:vassal.culture = {
						has_cultural_parameter = unruly_vaegirs
					}
				}
			}
			value = -30
		}
		else_if = {
			limit = {
				AND = {
					scope:liege ?= {
						highest_held_title_tier >= tier_kingdom
					}
					scope.liege.culture = {
						has_cultural_pillar = heritage_byalic
					}
					scope:liege.primary_title = {
						has_title_law_flag = elective_succession_law
					}
					scope:vassal.culture = {
						has_cultural_parameter = unruly_vaegirs
					}
				}
			}
			value = 15
		}
	}
}
