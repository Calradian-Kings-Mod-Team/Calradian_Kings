	music = "event:/DLC/FP2/MUSIC/moodtrack/mx_iberian_moodTrack1"
	group = group_foi

	is_valid = {
		OR = {
			culture = {
				has_cultural_pillar = heritage_nahasi

			culture = {
				any_parent_culture_or_above = { has_cultural_pillar = heritage_nahasi }
			}
			culture = {
